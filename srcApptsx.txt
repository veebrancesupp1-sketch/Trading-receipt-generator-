import { useEffect, useRef, useState } from 'react';
     import { AppState, GlobalSettings, Trade } from './types';
     import { TradeForm } from './components/TradeForm';
     import { ReceiptPreview } from './components/ReceiptPreview';
     import { ThemeToggle } from './components/ThemeToggle';
     import { ExportButtons } from './components/ExportButtons';                                                                                                                      import { Toast, useToast } from './components/Toast';
     import { getDefaultState, loadFromUrl } from './utils/serialization';

     function App() {
       const [state, setState] = useState<AppState>(() => {
         const urlState = loadFromUrl();
         return urlState || getDefaultState();
       });
       const { toasts, removeToast } = useToast();
       const receiptRef = useRef<HTMLDivElement>(null);
       const [isMobile, setIsMobile] = useState(false);
                                                                                                                                                                                        useEffect(() => {                                                                                                                                                                  const checkMobile = () => setIsMobile(window.innerWidth < 768);
         checkMobile();
         window.addEventListener('resize', checkMobile);
         return () => window.removeEventListener('resize', checkMobile);
       }, []);                                                                                                                                                                   
       useEffect(() => {
         document.documentElement.classList.toggle('dark', state.theme === 'dark');
         document.body.classList.toggle('dark', state.theme === 'dark');
       }, [state.theme]);

       const toggleTheme = () => {
         setState(prev => ({ ...prev, theme: prev.theme === 'dark' ? 'light' : 'dark' }));
       };

       const handleSettingsChange = (settings: GlobalSettings) => {
         setState(prev => ({ ...prev, settings }));
       };

       const handleTradesChange = (trades: Trade[]) => {
         setState(prev => ({ ...prev, trades }));
       };

       const isDark = state.theme === 'dark';

       return (
         <div className={`min-h-screen ${isDark ? 'bg-slate-900' : 'bg-slate-100'}`}>
           <header className={`sticky top-0 z-40 backdrop-blur-md ${
             isDark ? 'bg-slate-900/90 border-slate-700' : 'bg-white/90 border-slate-200'
           } border-b`}>
             <div className="max-w-7xl mx-auto px-4 py-3 flex items-center justify-between">
               <div className="flex items-center gap-3">
                 <span className="text-2xl">📊</span>
                 <h1 className={`text-xl font-bold ${isDark ? 'text-white' : 'text-slate-800'}`}>
                   Trading Receipt
                 </h1>
               </div>
               <ThemeToggle theme={state.theme} onToggle={toggleTheme} />
             </div>
           </header>

           <main className="max-w-7xl mx-auto p-4">
             <div className="flex flex-col lg:flex-row gap-6">
               <div className="lg:w-1/2">
                 <TradeForm
                   settings={state.settings}
                   trades={state.trades}
                   theme={state.theme}
                   onSettingsChange={handleSettingsChange}
                   onTradesChange={handleTradesChange}
                 />
               </div>

               <div className="lg:w-1/2">
                 <div className="lg:sticky lg:top-20">
                   <div className="mb-4">
                     <ExportButtons receiptRef={receiptRef} state={state} />
                   </div>
                   <ReceiptPreview
                     accountName={state.settings.accountName}
                     baseCurrency={state.settings.baseCurrency}
                     startingBalance={state.settings.startingBalance}
                     trades={state.trades}
                     theme={state.theme}
                     receiptRef={receiptRef}
                   />
                 </div>
               </div>
             </div>
           </main>

           <Toast toasts={toasts} removeToast={removeToast} />

           <style>{`
             @keyframes slide-in {
               from {
                 opacity: 0;
                 transform: translateX(100%);
               }
               to {
                 opacity: 1;
                 transform: translateX(0);
               }
             }
             .animate-slide-in {
               animation: slide-in 0.3s ease;
             }
           `}</style>
         </div>
       );
     }

     export default App;
